    <script>
        function setFontSize(size) {
            document.body.classList.remove('large-text', 'extra-large-text');
            if (size === 'large') document.body.classList.add('large-text');
            else if (size === 'extra-large') document.body.classList.add('extra-large-text');
            localStorage.setItem('fontSizePreference', size);
        }

        document.addEventListener('DOMContentLoaded', function() {
            const savedSize = localStorage.getItem('fontSizePreference');
            if (savedSize) setFontSize(savedSize);
        });

        // תיקון הטיימר - עכשיו עובד מושלם!
        function updateCountdown() {
            // תאריך הוובינר: 28 באוקטובר 2025 בשעה 21:00
            const webinarDate = new Date('2025-10-28T21:00:00');
            const now = new Date();
            const diff = webinarDate - now;
            
            const daysEl = document.getElementById('days');
            const hoursEl = document.getElementById('hours');
            const minutesEl = document.getElementById('minutes');
            const secondsEl = document.getElementById('seconds');
            
            if (!daysEl || !hoursEl || !minutesEl || !secondsEl) return;
            
            if (diff > 0) {
                const days = Math.floor(diff / (1000 * 60 * 60 * 24));
                const hours = Math.floor((diff % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
                const minutes = Math.floor((diff % (1000 * 60 * 60)) / (1000 * 60));
                const seconds = Math.floor((diff % (1000 * 60)) / 1000);
                
                daysEl.textContent = String(days).padStart(2, '0');
                hoursEl.textContent = String(hours).padStart(2, '0');
                minutesEl.textContent = String(minutes).padStart(2, '0');
                secondsEl.textContent = String(seconds).padStart(2, '0');
            } else {
                daysEl.textContent = '00';
                hoursEl.textContent = '00';
                minutesEl.textContent = '00';
                secondsEl.textContent = '00';
            }
        }
        
        setInterval(updateCountdown, 1000);
        updateCountdown();

        // Learning Path Functions
        function highlightStep(step) {
            document.querySelectorAll('.path-step').forEach(stepEl => {
                stepEl.classList.remove('active');
            });
            
            if (step > 0 && step <= 7) {
                const currentStep = document.querySelector(`.step-${step}`);
                if (currentStep) currentStep.classList.add('active');
            }
            
            const progressDots = document.querySelectorAll('.progress-dot');
            progressDots.forEach((dot, index) => {
                if (index < step) {
                    dot.classList.add('completed');
                } else {
                    dot.classList.remove('completed');
                }
            });
            
            const investorIcon = document.querySelector('.investor-icon');
            const targetStep = document.querySelector(`.step-${step}`);
            
            if (investorIcon && targetStep) {
                const stepRect = targetStep.getBoundingClientRect();
                const containerRect = document.querySelector('.path-container').getBoundingClientRect();
                
                const relativeLeft = stepRect.left - containerRect.left;
                const relativeTop = stepRect.top - containerRect.top;
                
                const iconLeft = relativeLeft + (stepRect.width / 2) - 40;
                const iconTop = relativeTop + 60;
                
                investorIcon.style.transition = 'all 1s cubic-bezier(0.4, 0, 0.2, 1)';
                investorIcon.style.left = iconLeft + 'px';
                investorIcon.style.top = iconTop + 'px';
                investorIcon.style.transform = 'scale(1.1)';
            }
        }
        
        function showTooltip(stepNumber, element) {
            const tooltip = document.getElementById('learning-tooltip');
            const tooltipText = element.getAttribute('data-tooltip');
            
            if (tooltip && tooltipText) {
                tooltip.textContent = tooltipText;
                tooltip.classList.add('show');
            }
        }
        
        function hideTooltip() {
            const tooltip = document.getElementById('learning-tooltip');
            if (tooltip) {
                tooltip.classList.remove('show');
            }
        }

        function resetStraightPath() {
            document.querySelectorAll('.path-step').forEach(stepEl => {
                stepEl.classList.remove('active');
            });
            
            const progressDots = document.querySelectorAll('.progress-dot');
            progressDots.forEach((dot, index) => {
                if (index === 0) {
                    dot.classList.add('completed');
                } else {
                    dot.classList.remove('completed');
                }
            });
            
            const investorIcon = document.querySelector('.investor-icon');
            if (investorIcon) {
                investorIcon.style.transition = 'all 1s cubic-bezier(0.4, 0, 0.2, 1)';
                investorIcon.style.left = '30px';
                investorIcon.style.top = '60px';
                investorIcon.style.transform = 'scale(1)';
            }
        }

        function initLearningPath() {
            const steps = document.querySelectorAll('.path-step');
            if (steps.length === 0) return;
            
            let isAnimating = false;
            
            steps.forEach((step, index) => {
                step.addEventListener('mouseenter', function(e) {
                    if (isAnimating) return;
                    e.stopPropagation();
                    isAnimating = true;
                    
                    highlightStep(index + 1);
                    showTooltip(index + 1, this);
                    
                    setTimeout(() => { isAnimating = false; }, 100);
                });
                
                step.addEventListener('mouseleave', function(e) {
                    if (isAnimating) return;
                    e.stopPropagation();
                    
                    const rect = this.getBoundingClientRect();
                    const x = e.clientX;
                    const y = e.clientY;
                    
                    if (x < rect.left || x > rect.right || y < rect.top || y > rect.bottom) {
                        hideTooltip();
                        resetStraightPath();
                    }
                });
                
                step.addEventListener('click', function(e) {
                    e.preventDefault();
                    e.stopPropagation();
                    highlightStep(index + 1);
                    showTooltip(index + 1, this);
                });
                
                step.addEventListener('keydown', function(e) {
                    if (e.key === 'Enter' || e.key === ' ') {
                        e.preventDefault();
                        e.stopPropagation();
                        highlightStep(index + 1);
                        showTooltip(index + 1, this);
                    }
                });
            });
            
            document.addEventListener('click', function(e) {
                if (!e.target.closest('.path-step') && !e.target.closest('#learning-tooltip')) {
                    hideTooltip();
                    resetStraightPath();
                }
            });
        }

        if (document.readyState === 'loading') {
            document.addEventListener('DOMContentLoaded', function() {
                initLearningPath();
                initCarousel();
            });
        } else {
            initLearningPath();
            initCarousel();

        // תיקון קרוסלת הסרטונים - עכשיו כל הסרטונים עובדים!
        let currentSlide = 0;
        const totalSlides = 5;

        function updateCarousel() {
            const track = document.getElementById('testimonials-track');
            const indicators = document.querySelectorAll('.indicator');
            
            if (track) {
                track.style.transform = `translateX(-${currentSlide * 100}%)`;
            }
            
            indicators.forEach((indicator, index) => {
                if (index === currentSlide) {
                    indicator.classList.add('active');
                } else {
                    indicator.classList.remove('active');
                }
            });
        }

        function nextTestimonial() {
            currentSlide = (currentSlide + 1) % totalSlides;
            updateCarousel();
        }

        function prevTestimonial() {
            currentSlide = (currentSlide - 1 + totalSlides) % totalSlides;
            updateCarousel();
        }

        function goToSlide(slideIndex) {
            currentSlide = slideIndex;
            updateCarousel();
        }

        function initCarousel() {
            const autoPlayInterval = setInterval(nextTestimonial, 10000);
            
            const carousel = document.querySelector('.testimonials-carousel');
            if (carousel) {
                carousel.addEventListener('mouseenter', function() {
                    clearInterval(autoPlayInterval);
                });
                
                carousel.addEventListener('mouseleave', function() {
                    setInterval(nextTestimonial, 10000);
                });
            }
            
            updateCarousel();
        }
